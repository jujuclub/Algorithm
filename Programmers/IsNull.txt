-- 이름이 없는 동물의 아이디
SELECT ANIMAL_ID
FROM ANIMAL_INS
WHERE NAME IS NULL;

-- 이름이 있는 동물의 아이디
SELECT ANIMAL_ID
FROM ANIMAL_INS
WHERE NAME IS NOT NULL
ORDER BY ANIMAL_ID;

-- NULL 처리하기 : null부분만 어떻게 값을 바꿔야 할지 모르겠다
SELECT ANIMAL_TYPE, NAME, SEX_UPON_INTAKE
FROM ANIMAL_INS
WHERE NAME IS NULL AS 'No name'
ORDER BY ANIMAL_ID;
